def list_folders():
    """åˆ—å‡º 'æ­¤ç”µè„‘' é‡Œçš„è‡ªå®šä¹‰æ–‡ä»¶å¤¹"""
    try:
        with winreg.OpenKey(winreg.HKEY_LOCAL_MACHINE, NAMESPACE_PATH) as key:
            index = 0

                    subkey_name = winreg.EnumKey(key, index)  # è·å– GUID
                    guid = subkey_name.strip("{}")

                    # è·å–æ–‡ä»¶å¤¹åç§°
                    try:

                    index += 1
                except OSError:
                    break

                print("âš ï¸ No custom folders found in 'This PC'.")

    except PermissionError:
        print("âŒ Error: This script requires administrator privileges.")
        print("ğŸ”¹ Try running the script as an administrator.")
